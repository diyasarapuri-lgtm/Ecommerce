// Custom hook to initialize Firebase and manage Auth state
const useFirebase = () => {
    const [db, setDb] = useState(null);
    const [auth, setAuth] = useState(null);
    const [userId, setUserId] = useState(null);
    const [isAuthReady, setIsAuthReady] = useState(false);

    useEffect(() => {
        if (!firebaseConfig) {
            console.error("Firebase config is missing.");
            setIsAuthReady(true); // Treat as ready but unauthenticated if config is missing
            return;
        }

        try {
            const app = initializeApp(firebaseConfig);
            const firestore = getFirestore(app);
            const authInstance = getAuth(app);

            setDb(firestore);
            setAuth(authInstance);

            const unsubscribe = onAuthStateChanged(authInstance, async (user) => {
                if (!user && initialAuthToken) {
                    // Sign in with custom token if available
                    try {
                        await signInWithCustomToken(authInstance, initialAuthToken);
                        // onAuthStateChanged will be called again with the signed-in user
                    } catch (error) {
                        console.error("Custom token sign-in failed:", error);
                        // Fallback to anonymous sign-in if custom token fails
                        try {
                            const anonymousUser = await signInAnonymously(authInstance);
                            setUserId(anonymousUser.user.uid);
                        } catch (anonError) {
                            console.error("Anonymous sign-in failed:", anonError);
                        }
                    }
                } else if (!user) {
                    // Sign in anonymously if no custom token
                    try {
                        const anonymousUser = await signInAnonymously(authInstance);
                        setUserId(anonymousUser.user.uid);
                    } catch (error) {
                        console.error("Anonymous sign-in failed:", error);
                    }
                } else {
                    // User is signed in
                    setUserId(user.uid);
                }
                setIsAuthReady(true);
            });

            return () => unsubscribe();
        } catch (error) {
            console.error("Firebase initialization failed:", error);
            setIsAuthReady(true);
        }
    }, []);

    return { db, auth, userId, isAuthReady };
};


// Custom hook to manage Firestore cart data
const useCartStore = (db, userId, isAuthReady) => {
    const [cartItems, setCartItems] = useState([]);
    const [isLoading, setIsLoading] = useState(true);
    const [error, setError] = useState(null);

    // Cart path helper
    const getCartCollection = useCallback(() => {
        if (db && userId) {
            // Using a user-specific path in the artifacts public area for persistence
            // This path structure is safe under the provided Canvas security rules.
            return collection(db, 'artifacts', appId, 'users', userId, 'diligent_cart');
        }
        return null;
    }, [db, userId]);

    // Setup real-time listener for the cart
    useEffect(() => {
        const cartCollectionRef = getCartCollection();

        if (!isAuthReady || !cartCollectionRef) {
            if (isAuthReady) setIsLoading(false); // Auth is ready, but no cart ref (e.g., no userId yet)
            return;
        }

        setIsLoading(true);
        setError(null);

        const unsubscribe = onSnapshot(
            cartCollectionRef,
            (snapshot) => {
                const items = snapshot.docs.map(doc => ({
                    id: doc.id,
                    ...doc.data(),
                    // Ensure price is treated as a number
                    price: Number(doc.data().price)
                }));
                setCartItems(items);
                setIsLoading(false);
            },
            (err) => {
                console.error("Error fetching cart:", err);
                setError("Failed to load cart items.");
                setIsLoading(false);
            }
        );

        return () => unsubscribe();
    }, [getCartCollection, isAuthReady]);

    // Action: Add/Increment item in cart
    const addToCart = useCallback(async (product, quantity = 1) => {
        const cartCollectionRef = getCartCollection();
        if (!cartCollectionRef) return;

        const cartItemDocRef = doc(cartCollectionRef, product.id);

        try {
            await runTransaction(db, async (transaction) => {
                const docSnap = await transaction.get(cartItemDocRef);

                if (docSnap.exists()) {
                    // Item exists, increment quantity
                    const currentQuantity = docSnap.data().quantity;
                    const newQuantity = currentQuantity + quantity;
                    transaction.update(cartItemDocRef, { quantity: newQuantity });
                } else {
                    // Item does not exist, create new
                    transaction.set(cartItemDocRef, {
                        productId: product.id,
                        name: product.name,
                        price: product.price,
                        imageUrl: product.imageUrl,
                        quantity: quantity,
                    });
                }
            });
        } catch (e) {
            console.error("Transaction failed: ", e);
        }
    }, [getCartCollection, db]);

    // Action: Update quantity or remove
    const updateCartQuantity = useCallback(async (productId, newQuantity) => {
        const cartCollectionRef = getCartCollection();
        if (!cartCollectionRef) return;

        const cartItemDocRef = doc(cartCollectionRef, productId);

        if (newQuantity <= 0) {
            // Remove item if quantity is zero or less
            try {
                await deleteDoc(cartItemDocRef);
            } catch (e) {
                console.error("Error removing item:", e);
            }
        } else {
            // Update quantity
            try {
                await setDoc(cartItemDocRef, { quantity: newQuantity }, { merge: true });
            } catch (e) {
                console.error("Error updating item:", e);
            }
        }
    }, [getCartCollection]);

    // Action: Remove item completely
    const removeFromCart = useCallback(async (productId) => {
        const cartCollectionRef = getCartCollection();
        if (!cartCollectionRef) return;

        const cartItemDocRef = doc(cartCollectionRef, productId);
        try {
            await deleteDoc(cartItemDocRef);
        } catch (e) {
            console.error("Error removing item:", e);
        }
    }, [getCartCollection]);

    // Memoize total calculation
    const cartTotal = useMemo(() =>
        cartItems.reduce((total, item) => total + item.price * item.quantity, 0),
        [cartItems]
    );

    const cartCount = useMemo(() =>
        cartItems.reduce((count, item) => count + item.quantity, 0),
        [cartItems]
    );

    return {
        cartItems,
        cartTotal,
        cartCount,
        isLoading,
        error,
        addToCart,
        updateCartQuantity,
        removeFromCart,
        isReady: isAuthReady && !isLoading // True when Auth is done AND data is loaded
    };
};

// --- Main Application Component ---
const App = () => {
    const { db, userId, isAuthReady } = useFirebase();
    const cartState = useCartStore(db, userId, isAuthReady);

    const [isCartOpen, setIsCartOpen] = useState(false);
    const [selectedProduct, setSelectedProduct] = useState(null);
    const [view, setView] = useState('list'); // 'list' or 'details'

    const handleViewDetails = (product) => {
        setSelectedProduct(product);
        setView('details');
    };

    const handleCloseDetails = () => {
        setSelectedProduct(null);
        setView('list');
    };

    const handleAddToCart = (product) => {
        if (!cartState.isReady) {
            console.error("Cart system not ready. Cannot add item.");
            return;
        }
        cartState.addToCart(product);
        setIsCartOpen(true);
    };

    // Show a loading/setup screen until Firebase/Firestore is ready
    if (!isAuthReady) {
        return (
            <div className="flex items-center justify-center min-h-screen bg-gray-50">
                <Loader2 className="w-8 h-8 animate-spin text-indigo-600" />
                <p className="ml-3 text-lg font-medium text-indigo-700">Initializing Application...</p>
            </div>
        );
    }

    return (
        <div className="min-h-screen bg-gray-50 font-sans">
            <Header
                onToggleCart={() => setIsCartOpen(!isCartOpen)}
                cartCount={cartState.cartCount}
                userId={userId}
            />

            <main className="pt-20 pb-8 container mx-auto px-4 max-w-7xl">
                {view === 'list' && (
                    <section>
                        <h2 className="text-3xl font-extrabold text-gray-900 mb-6">Featured Products</h2>
                        <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 gap-6">
                            {MOCK_PRODUCTS.map(product => (
                                <ProductCard
                                    key={product.id}
                                    product={product}
                                    onAddToCart={handleAddToCart}
                                    onViewDetails={handleViewDetails}
                                />
                            ))}
                        </div>
                    </section>
                )}

                {view === 'details' && selectedProduct && (
                    <section className="bg-white p-6 rounded-xl shadow-lg">
                        <button
                            onClick={handleCloseDetails}
                            className="text-indigo-600 hover:text-indigo-800 flex items-center mb-6"
                        >
                            <ArrowLeft className="w-5 h-5 mr-1" /> Back to Products
                        </button>
                        <div className="flex flex-col md:flex-row gap-8">
                            <div className="md:w-1/2">
                                <img
                                    src={selectedProduct.imageUrl}
                                    alt={selectedProduct.name}
                                    className="w-full h-auto object-cover rounded-xl shadow-md"
                                    onError={(e) => { e.target.onerror = null; e.target.src = 'https://placehold.co/600x400/e0e0e0/333?text=Detail+View' }}
                                />
                            </div>
                            <div className="md:w-1/2">
                                <span className="text-sm font-semibold uppercase text-indigo-600 mb-2 block">{selectedProduct.category}</span>
                                <h2 className="text-4xl font-bold text-gray-900 mb-3">{selectedProduct.name}</h2>
                                <p className="text-3xl font-extrabold text-indigo-700 mb-6">${selectedProduct.price.toFixed(2)}</p>
                                <p className="text-gray-700 mb-8 leading-relaxed">{selectedProduct.description}</p>
                                <button
                                    onClick={() => handleAddToCart(selectedProduct)}
                                    className="w-full md:w-auto flex items-center justify-center px-8 py-4 text-lg font-semibold text-white bg-green-600 rounded-lg shadow-lg hover:bg-green-700 transition-colors"
                                >
                                    <ShoppingCart className="w-5 h-5 mr-3" />
                                    Add to Cart
                                </button>
                            </div>
                        </div>
                    </section>
                )}
            </main>

            <CartSidebar
                isCartOpen={isCartOpen}
                onClose={() => setIsCartOpen(false)}
                cartState={cartState}
            />
        </div>
    );
};

export default App;
